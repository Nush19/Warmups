Return an int array length n containing the first n digits of pi.

MakePi(3) -> {3, 1, 4}

public int[] MakePi(int n) {
    double pi = Math.PI;
    var str = pi.ToString().Remove(1,1);
    var numbers = new int[n];

    for(int i = 0; i < n; i++)
    {
        numbers[i] = int.Parse(str[i].ToString());
    }
    return numbers;
}
